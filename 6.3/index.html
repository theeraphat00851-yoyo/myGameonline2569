            // วาดผู้เล่นทุกคนบนโต๊ะเรียนประจำฝั่งเรียลไทม์!

            activePlayersList.forEach(p => {

                if (p.team !== 'teacher') {

                    drawPlayer(p);

                }

            });


            // อนิเมชันสายยิงปาสมุดดินสอเกรียดชังใส่กัน

            let activeProjectiles = [];

            projectiles.forEach((proj) => {

                proj.progress += proj.speed;


                const currentX = proj.x + (proj.targetX - proj.x) * proj.progress;

                const baseHeight = proj.y + (proj.targetY - proj.y) * proj.progress;

                const heightOffset = -140 * Math.sin(proj.progress * Math.PI);

                const currentY = baseHeight + heightOffset;


                // หางควันประกายไฟสติ๊กเกอร์พ่นพิกเซล

                ctx.beginPath();

                ctx.arc(currentX, currentY, (proj.size / 2) - 3, 0, Math.PI * 2);

                ctx.fillStyle = proj.color;

                ctx.fill();


                ctx.font = `${proj.size}px Inter`;

                ctx.fillText(proj.emoji, currentX - (proj.size / 2), currentY + (proj.size / 2.5));


                if (proj.progress >= 1.0) {

                    playSound('hit');

                    createExplosion(proj.targetX, proj.targetY, proj.color);

                    createFloatingText(proj.targetX, proj.targetY - 40, `-${proj.damage} DMG`, proj.team === 'cat' ? '#f43f5e' : '#06b6d4');

                } else {

                    activeProjectiles.push(proj);

                }

            });

            projectiles = activeProjectiles;


            // เอฟเฟกต์สะเก็ดระเบิดชอล์กพิกเซล

            let activeParticles = [];

            particles.forEach((p) => {

                p.x += p.vx;

                p.y += p.vy;

                p.vy += 0.13; 

                p.alpha -= p.decay;


                if (p.alpha > 0) {

                    ctx.save();

                    ctx.globalAlpha = p.alpha;

                    ctx.beginPath();

                    ctx.arc(p.x, p.y, p.radius, 0, Math.PI * 2);

                    ctx.fillStyle = p.color;

                    ctx.fill();

                    ctx.restore();

                    activeParticles.push(p);

                }

            });

            particles = activeParticles;


            // เอฟเฟกต์ตัวเลขลอยเด้งหยาบๆ

            let activeTexts = [];

            damageTexts.forEach((t) => {

                t.y += t.vy;

                t.alpha -= 0.019;


                if (t.alpha > 0) {

                    ctx.save();

                    ctx.globalAlpha = t.alpha;

                    ctx.fillStyle = t.color;

                    ctx.font = "black 20px Sriracha";

                    ctx.shadowColor = "#000000";

                    ctx.shadowBlur = 4;

                    ctx.fillText(t.text, t.x - 25, t.y);

                    ctx.restore();

                    activeTexts.push(t);

                }

            });

            damageTexts = activeTexts;


            requestAnimationFrame(gameLoop);

        }
